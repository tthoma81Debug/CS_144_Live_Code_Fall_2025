import java.util.Scanner;

public class LabProgram {
   public static void main(String[] args) {
   Scanner scnr = new Scanner(System.in);
   int enteredNum = 9999;
   int max = -1;
   double average;
   int counter = 0;
   int numberOfItems;
   int sum = 0;


   while(enteredNum >= 0)
   {
      enteredNum = scnr.nextInt();

      //don't add if last num is -1
      if(enteredNum < 0)
      {
         break;
      }

      if(enteredNum > max)
      {
         max = enteredNum;
      }

      counter++;
      sum+= enteredNum;
   }

   numberOfItems = counter;

   //more explicit casting
   double decimalSum = (double)sum;
   double decimalNumberOfItems = (double)numberOfItems;

   average = decimalSum / decimalNumberOfItems;


   //debugging
   //System.out.println("Sum is " + decimalSum);
   //System.out.println("Total Number of Items is " + numberOfItems);

   System.out.print(max + " ");
   System.out.printf("%.2f\n", average);


   }
}
